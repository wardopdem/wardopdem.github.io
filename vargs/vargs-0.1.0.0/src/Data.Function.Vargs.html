module Data.Function.Vargs (
    -- * ╨д╤Г╨╜╨║╤Ж╨╕╤П ╨┤╨╗╤П ╤А╨╡╨░╨╗╨╕╨╖╨░╤Ж╨╕╨╕ ╨┐╨╡╤А╨╡╨╝╨╡╨╜╨╜╨│╨╛ ╤З╨╕╤Б╨╗╨░ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓.
    defVargsFun,
    -- * ╨Ъ╨╗╨░╤Б╤Б╤Л ╨╕ ╤В╨╕╨┐╤Л ╨┤╨╗╤П ╤А╨╡╨░╨╗╨╕╨╖╨░╤Ж╨╕╨╕ ╨╝╨╡╤Е╨░╨╜╨╕╨╖╨╝╨░ ╨┐╨╡╤А╨╡╨╝╨╡╨╜╨╜╨╛╨│╨╛ ╤З╨╕╤Б╨╗╨░ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓
    InstMaker, InstMakerQ, InstSrc(..),  ArgProc(..), Genz(..)
) where 

import Language.Haskell.TH
import Language.Haskell.TH.Syntax
import qualified Data.Map as M

-- | ╨У╨╡╨╜╨╡╤А╨░╤В╨╛╤А ╨┤╨╡╨║╨╗╨░╤А╨░╤Ж╨╕╨╣ ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╨╛╨▓ [`InstanceD']
type InstMaker = Type -> Type ->  Name -> DecsQ   
-- ^ * :: ╨║╨╗╨░╤Б╤Б 
-- * -> ╤В╨╕╨┐ 
-- * -> ╨╝╨╡╤В╨╛╨┤ 
-- * -> ╨┤╨╡╨║╨╗╨░╤А╨░╤Ж╨╕╨╕ ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╨╛╨▓

-- | InstMaker ╨╖╨░╨▓╤С╤А╤В╨╜╤Г╤В╤Л╨╣ ╨▓ ╨╝╨░╨╜╨░╨┤╤Г ╤Ж╨╕╤В╨╕╤А╨╛╨▓╨░╨╜╨╕╤П Q
type InstMakerQ = Q InstMaker

-- | ╨Ъ╨╗╨░╤Б╤Б ╤В╨╕╨┐╨╛╨▓, ╨╜╨░ ╨╛╤Б╨╜╨╛╨▓╨╡ ╨║╨╛╤В╨╛╤А╨╛╨│╨╛ ╨╝╨╛╨╢╨╜╨╛ ╨┐╨╛╤Б╤В╤А╨╛╨╕╤В╤М ╨┐╤А╨╛╤Ж╨╡╨┤╤Г╤А╤Г ╨│╨╡╨╜╨╡╤А╨░╤Ж╨╕╨╕ ╨┤╨╡╨║╨╗╨░╤А╨░╤Ж╨╕╨╣ ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╨╛╨▓ [`InstanceD']
class InstSrc a where
    toMaker :: a -> InstMakerQ -- ^ ╨У╨╡╨╜╨╡╤А╨░╤В╨╛╤А ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╨╛╨▓ ╨┤╨╗╤П ╨╖╨░╨┤╨░╨╜╨╜╨╛╨│╨╛ ╤В╨╕╨┐╨░

mkTpHndl :: Cxt -> Type -> Exp -> InstMakerQ
mkTpHndl c t e = return $ \cnt_as cnt_at mt -> 
    return [InstanceD Nothing c (AppT cnt_as t) [ValD (VarP mt) (NormalB e) []]]

-- | ╨У╨╡╨╜╨╡╤А╨░╤В╨╛╤А ╨┤╨╡╨║╨╗╨░╤А╨░╤Ж╨╕╨╣ ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╨╛╨▓ ╨┐╨╛ ╨┐╨░╤А╨░╨╝ ╨▓╨╕╨┤╨░: @(''Type,  [| expr |])@
instance InstSrc (Name, ExpQ) where
    toMaker (n, e) = do 
        e' <- e 
        mkTpHndl [] (ConT n) e'

-- | ╨У╨╡╨╜╨╡╤А╨░╤В╨╛╤А ╨┤╨╡╨║╨╗╨░╤А╨░╤Ж╨╕╨╣ ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╨╛╨▓ ╨┐╨╛ ╨┐╨░╤А╨░╨╝ ╨▓╨╕╨┤╨░: @([t| [Int] |],  [| expr |])@ 
instance InstSrc (TypeQ, ExpQ) where
    toMaker (t, e) = do 
        t' <- t 
        e' <- e 
        mkTpHndl [] t' e'

-- | ╨Ъ╨╛╨╝╨▒╨╕╨╜╨░╤В╨╛╤А ╨│╨╡╨╜╨╡╤А╨░╤В╨╛╤А╨╛╨▓ ╨┤╨╡╨║╨╗╨░╤А╨░╤Ж╨╕╨╣ ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╨╛╨▓: ╤Б╤В╤А╨╛╨╕╤В ╨╜╨╛╨▓╤Л╨╣ ╨│╨╡╨╜╨╡╤А╨░╤В╨╛╤А, 
-- ╨║╨╛╤В╨╛╤А╤Л╨╣ ╨▓╤Л╨┐╨╛╨╗╨╜╤П╨╡╤В ╨╖╨░╨┤╨░╨╜╨╜╤Л╨╡ ╤Б╨┐╨╕╤Б╨║╨╛╨╝ ╨│╨╡╨╜╨╡╤А╨░╤В╨╛╤А╤Л ╨╕ ╨╛╨▒╤К╨╡╨┤╨╕╨╜╤П╨╡╤В ╨╕╤Е ╤А╨╡╨╖╤Г╨╗╤М╤В╨░╤В╤Л 
concIms :: [InstMaker] -> InstMakerQ
concIms ims =  
    return $ \cnt_as cnt_at mt -> do
        decs <- sequenceQ $ map (\i -> i cnt_as cnt_at mt) ims 
        return $ concat decs
               
-- | ╨У╨╡╨╜╨╡╤А╨░╤В╨╛╤А ╨┤╨╡╨║╨╗╨░╤А╨░╤Ж╨╕╨╣ ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╨╛╨▓ ╨┐╨╛ ╨┐╨░╤А╨░╨╝ ╨▓╨╕╨┤╨░: @([''Type1, ''Type2, ...],  [| expr |])@ 
instance InstSrc ([Name], ExpQ) where
    toMaker (ns, e) = do 
        e' <- e
        ims <- sequenceQ [mkTpHndl [] (ConT n) e' | n <- ns]
        concIms ims

-- | ╨У╨╡╨╜╨╡╤А╨░╤В╨╛╤А ╨┤╨╡╨║╨╗╨░╤А╨░╤Ж╨╕╨╣ ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╨╛╨▓ ╨┐╨╛ ╤Б╨┐╨╕╤Б╨║╤Г ╨╕╨╝╤С╨╜ ╤В╨╕╨┐╨╛╨▓ ╨▓╨╕╨┤╨░ @[''Integer, ''Double, ...]@.
-- ╨Я╨╛╨┤╤А╨░╨╖╤Г╨╝╨╡╨▓╨░╨╡╤В╤Б╤П, ╤З╤В╨╛ ╤В╨╕╨┐ ╤Н╨╗╨╡╨╝╨╡╨╜╤В╨╛╨▓ ╤Б╨┐╨╕╤Б╨║╨░ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓ ╨╕╨╝╨╡╨╡╤В 
-- ╨╡╨┤╨╕╨╜╤Б╤В╨▓╨╡╨╜╨╜╤Л╨╣ ╨║╨╛╨╜╤Б╤В╤А╤Г╨║╤В╨╛╤А ╤Б ╨╡╨┤╨╕╨╜╤Б╤В╨▓╨╡╨╜╨╜╤Л╨╝ ╨░╤А╨│╤Г╨╝╨╡╨╜╤В╨╛╨╝ (╤В╨╕╨┐-╨╛╨▒╤С╤А╤В╨║╨░) - ╤Н╤В╨╛╤В ╨║╨╛╨╜╤Б╤В╤А╤Г╨║╤В╨╛╤А 
-- ╨╕ ╨╕╤Б╨┐╨╛╨╗╤М╨╖╤Г╨╡╤В╤Б╤П ╨▓ ╨║╨░╤З╨╡╤Б╤В╨▓╨╡ "╤В╨╡╨╗╨░" ╨╝╨╡╤В╨╛╨┤╨░ ╤Б╨╛╨╖╨┤╨░╨╜╨╕╤П ╨╖╨╜╨░╤З╨╡╨╜╨╕╨╣ ╨╕╨╖ ╨╕╤Б╤Е╨╛╨┤╨╜╤Л╤Е ╤В╨╕╨┐╨╛╨▓
instance InstSrc [Name] where
    toMaker ns = do
        ims <- sequenceQ [mkTpHndl' (ConT n) | n <- ns]
        concIms ims
            where mkTpHndl' t = return $ \cnt_as cnt_at mt -> do
                    wc  <- wrapCons cnt_at  
                    return [InstanceD Nothing [] (AppT cnt_as t) [ValD (VarP mt) (NormalB wc) []]]
                  wrapCons (ConT cn) = do
                    TyConI (DataD _ _ _ _ [ForallC _ _ (NormalC cn' _)] _) <- reify cn
                    conE cn'

-- | ╨Ю╨▒╤С╤А╤В╨║╨░ ╨┤╨╗╤П ╨┤╨╡╨║╨╗╨░╤А╨░╤Ж╨╕╨╕ ╨╜╨╡╨╛╨▒╤Е╨╛╨┤╨╕╨╝╨╛╤Б╤В╨╕ ╨│╨╡╨╜╨╡╤А╨░╤Ж╨╕╨╕ "╨╛╨▒╨╛╨▒╤Й╤С╨╜╨╜╨╛╨│╨╛" ╤Н╨║╨╖╨╡╨┐╨╗╤П╤А╨░ ╨▓╨╕╨┤╨░ 
--
-- @a ~ T1, b ~ T2 => C a b ...@
newtype Genz = Genz TypeQ

-- | ╨Ф╨╗╤П ╨┐╨░╤А ╨▓╨╕╨┤╨░  (Genz [t| ... |], [| expr |]).
-- ╨б╤В╤А╨╛╨╕╤В ╨│╨╡╨╜╨╡╤А╨░╤В╨╛╤А ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╨╛╨▓ ╨▓╨╕╨┤╨░ 
--  @a ~ T1, b ~ T2, ... => C a b ...@
-- ╨│╨┤╨╡ Ti - ╤В╨╕╨┐ ╨▓╨╕╨┤╨░  Tp T1 T2 ... (╨╜╨░╨┐╤А╨╕╨╝╨╡╤А Double -> Double -> Double)
instance InstSrc (Genz, ExpQ) where
    toMaker (Genz tq, e) = do 
        t <- tq 
        e' <- e
        let (ctx, t') = genzType t 
        mkTpHndl ctx t' e'
            where
              -- | ╨У╨╡╨╜╨╡╤А╨░╤Ж╨╕╤П ╨┤╨╗╤П ╤В╨╕╨┐╨░ ╨▓╨╕╨┤╨░: 
              --  
              --    Q T1 T2 ... 
              --
              --   ╨┐╨░╤А╤Л ╨▓╨╕╨┤╨░ (╨┐╤Б╨╡╨▓╨┤╨╛╨║╨╛╨┤):
              -- 
              --  ([a ~ T1, b ~ T2, ... ], Q a b ...)
              --
              -- ╤В.╨╡. ╨║╨╛╨╜╤В╨╡╨║╤Б╤В╨░ ╤В╨╕╨┐╨╛╨▓ ╨╕ ╨┐╨╛╨╗╨╕╨╝╨╛╤А╤Д╨╜╨╛╨│╨╛ ╤В╨╕╨┐╨░ ╨▓ ╤Н╤В╨╛╨╝ ╨║╨╛╨╜╤В╨╡╨║╤Б╤В╨╡.
              genzType :: Type -> ([Type], Type)
              genzType t@(ConT nm) = ([], t)
              genzType tp =
                  let (t, m, tvs) = work tp (M.empty :: M.Map Name String) tpVars 
                  in (map mkCtx (M.assocs m), t)   
                      where work (ConT nm) m tvs =
                              case M.lookup nm m of
                                  Just s -> (VarT $ mkName s, m, tvs)
                                  Nothing -> let (s : tvs') = tvs 
                                             in (VarT $ mkName s, M.insert nm s m, tvs')
                            work (AppT t1 t2) m tvs = 
                              let (t', m', tvs') = work t1 m tvs
                              in  let (t'', m'', tvs'') = work t2 m' tvs'
                                  in (AppT t' t'', m'', tvs'')
                            work t m tvs = (t, m, tvs)
                            mkCtx (nm, s) = AppT (AppT EqualityT (VarT $ mkName s)) (ConT nm)
                            tpVars = map (:[]) ['a'..'z']  

-- | ╨а╨╡╨░╨╗╨╕╨╖╨░╤Ж╨╕╤П ╨┐╨╡╤А╨╡╨╝╨╡╨╜╨╜╨╛╨│╨╛ ╤З╨╕╤Б╨╗╨░ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓ ╤Д╤Г╨╜╨║╤Ж╨╕╨╕ defVargsFun
class ArgProc a where
    -- | ╨а╨╡╨░╨╗╨╕╨╖╨░╤Ж╨╕╤П ╨╝╨╡╤Е╨░╨╜╨╕╨╖╨╝╨░ ╨╜╨░╨║╨╛╨┐╨╗╨╡╨╜╨╕╤П/╨╛╨▒╤А╨░╨▒╨╛╤В╨║╨╕ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓ ╤В╨╕╨┐╨░ `InstMakerQ'
    prc :: String       -- ^ ╨Ш╨╝╤П ╤Б╨╛╨╖╨┤╨░╨▓╨░╨╡╨╝╨╛╨╣ ╤Д╤Г╨╜╨║╤Ж╨╕╨╕-╨╛╨▒╤С╤А╤В╨║╨╕ ╨╜╨░╨┤ @prc@.     
        -> [Name]       -- ^ ╨б╨┐╨╕╤Б╨╛╨║ ╨╕╨╝╤С╨╜ ╤Д╤Г╨╜╨║╤Ж╨╕╨╣, ╨║╨╛╤В╨╛╤А╤Л╨╡ ╨▒╤Г╨┤╤Г╤В ╨╕╤Б╨┐╨╛╨╗╤М╨╖╨╛╨▓╨░╨╜╤Л ╨▓ ╤В╨╡╤А╨╝╨╕╨╜╨░╨╗╤М╨╜╤Л╤Е 
                        -- ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╨░╤Е (╨╛╤В╨┤╨╡╨╗╤М╨╜╤Л╨╣ ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А ╨┤╨╗╤П ╨║╨░╨╢╨┤╨╛╨╣ ╤Д╤Г╨╜╨║╤Ж╨╕╨╕). ╨Ъ╨░╨╢╨┤╨░╤П ╨╕╨╖ ╤Д╤Г╨╜╨║╤Ж╨╕╨╣ ╨┤╨╛╨╗╨╢╨╜╨░ ╨╕╨╝╨╡╤В╤М 
                        -- ╨╕╨┤╨╡╨╜╤В╨╕╤З╨╜╤Л╨╣ ╨╜╨░╨▒╨╛╤А ╤В╨╕╨┐╨╛╨▓ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓ ╨╕ ╤Г╨╜╨╕╨║╨░╨╗╤М╨╜╤Л╨╣ ╤В╨╕╨┐ ╨▓╨╛╨╖╨▓╤А╨░╤В╨░.
        -> [InstMakerQ] -- ^ ╨У╨╡╨╜╨╡╤А╨░╤В╨╛╤А╤Л ╨┤╨╡╨║╨╗╨░╤А╨░╤Ж╨╕╨╣ ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╨╛╨▓. 
        -> a           

-- | ╨в╨╡╤А╨╝╨╕╨╜╨░╨╗╤М╨╜╤Л╨╣ ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А - ╨╖╨░╨┐╤Г╤Б╨║╨░╨╡╤В ╨│╨╡╨╜╨╡╤А╨░╤Ж╨╕╤О ╨▓╤Б╨╡╤Е ╨╜╨╡╨╛╨▒╤Е╨╛╨┤╨╕╨╝╤Л╤Е ╨┤╨╡╨║╨╗╨░╤А╨░╤Ж╨╕╨╣ 
--  ╨┐╨╛ ╤Д╨╕╨║╤Б╨╕╤А╨╛╨▓╨░╨╜╨╜╤Л╨╝ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨░╨╝ ╨╕ ╤Б╨┐╨╕╤Б╨║╤Г ╨│╨╡╨╜╨╡╤А╨░╤В╨╛╤А╨╛╨▓ ╨┤╨╡╨║╨╗╨░╤А╨░╤Ж╨╕╨╣ ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╨╛╨▓ (`InstMaker')
instance ArgProc DecsQ where
    prc = defVargsFun' 

-- | "╨Ь╨░╨│╨╕╤П" ╨┐╨╡╤А╨╡╨╝╨╡╨╜╨╜╨╛╨│╨╛ ╤З╨╕╤Б╨╗╨░ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓
instance (InstSrc a, ArgProc r) => ArgProc (a -> r) where
    -- | ╨Э╨░╨║╨░╨┐╨╗╨╕╨▓╨░╨╡╨╝ ╨│╨╡╨╜╨╡╤А╨░╤В╨╛╤А╤Л ╨┤╨╡╨║╨╗╨░╤А╨░╤Ж╨╕╨╣ ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╨╛╨▓ ╨┤╨╗╤П ╨┐╨╛╤Б╨╗╨╡╨┤╤Г╤О╤Й╨╡╨╣ ╨╛╨▒╤А╨░╨▒╨╛╤В╨║╨╕ 
    -- ╨▓ @prc@ ╨╕╨╖ ╤В╨╡╤А╨╝╨╕╨╜╨░╨╗╤М╨╜╨╛╨│╨╛ ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╨░ ArgProc DecsQ
    prc fn e sts = prc fn e . (: sts) . toMaker

-- | ╨У╨╡╨╜╨╡╤А╨░╤В╨╛╤А ╨┤╨╡╨║╨╗╨░╤А╨░╤Ж╨╕╨╣ ╨┤╨╗╤П ╤А╨╡╨░╨╗╨╕╨╖╨░╤Ж╨╕╨╕ ╨┐╨╡╤А╨╡╨╝╨╡╨╜╨╜╨╛╨│╨╛ ╤З╨╕╤Б╨╗╨░

defVargsFun :: ArgProc a 
            => String -- ^ ╨Ш╨╝╤П ╨┤╨╗╤П ╤Б╨╛╨╖╨┤╨░╨▓╨░╨╡╨╝╨╛╨╣ ╤Д╤Г╨╜╨║╤Ж╨╕╨╕  
            -> [Name] -- ^ * ╨б╨┐╨╕╤Б╨╛╨║ ╨╕╨╝╤С╨╜ ╤Д╤Г╨╜╨║╤Ж╨╕╨╣, ╨╜╨░ ╨╛╤Б╨╜╨╛╨▓╨╡ ╨║╨╛╤В╨╛╤А╤Л╤Е ╤Б╤В╤А╨╛╨╕╤В╤Б╤П ╤Д╤Г╨╜╨║╤Ж╨╕╤П ╤Б ╨┐╨╡╤А╨╡╨╝╨╡╨╜╨╜╤Л╨╝ ╤З╨╕╤Б╨╗╨╛╨╝ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓
                      -- ╨▓╤Б╨╡ ╤Д╤Г╨╜╨║╤Ж╨╕╨╕ ╨┤╨╛╨╗╨╢╨╜╤Л: 
                      --  * - ╨╕╨╝╨╡╤В╤М ╨┐╤А╨╛╤Б╤В╤Л╨╡ (╨╜╨╡ ╨┐╨╛╨╗╨╕╨╝╨╛╤А╤Д╨╜╤Л╨╡) ╤В╨╕╨┐╤Л ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓ ╨╕ ╤А╨╡╨╖╤Г╨╗╤М╤В╨░╤В╨░;
                      --  * - ╨╕╨╝╨╡╤В╤М ╨╕╨┤╨╡╨╜╤В╨╕╤З╨╜╤Л╨╡ ╤В╨╕╨┐╤Л ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓
                      --  * - ╨╕╨╝╨╡╤В╤М ╤Г╨╜╨╕╨║╨░╨╗╤М╨╜╤Л╨╣ ╤В╨╕╨┐ ╨▓╨╛╨╖╨▓╤А╨░╤В╨░
            -> a      -- ^ ╨Ф╨░╨╗╨╡╨╡ ╤Б╨╗╨╡╨┤╤Г╨╡╤В ╨┐╤А╨╛╨╕╨╖╨▓╨╛╨╗╤М╨╜╨╛╨╡ ╨║╨╛╨╗╨╕╤З╨╡╤Б╤В╨▓╨╛ ╨╖╨╜╨░╤З╨╡╨╜╨╕q ╤В╨╕╨┐╨╛╨▓ ╨║╨╗╨░╤Б╤Б╨░ `InstSrc' 
defVargsFun fn sfns  = prc fn sfns []

-- | ╨д╨░╨║╤В╨╕╤З╨╡╤Б╨║╨░╤П  ╨│╨╡╨╜╨╡╤А╨░╤Ж╨╕╤П ╨┤╨╡╨║╨╗╨░╤А╨░╤Ж╨╕╨╣ ╨┤╨╗╤П ╤А╨╡╨░╨╗╨╕╨╖╨░╤Ж╨╕╨╕ ╨┐╨╡╤А╨╡╨╝╨╡╨╜╨╜╨╛╨│╨╛ ╤З╨╕╤Б╨╗╨░ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓
defVargsFun' :: String        -- ^ ╨Ш╨╝╤П ╨┤╨╗╤П ╤Б╨╛╨╖╨┤╨░╨▓╨░╨╡╨╝╨╛╨╣ ╤Д╤Г╨╜╨║╤Ж╨╕ ╤Б ╨┐╨╡╤А╨╡╨╝╨╡╨╜╨╜╤Л╨╝ ╤З╨╕╤Б╨╗╨╛╨╝ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓ (╨┤╨░╨╗╨╡╨╡, ╨д╤Г╨╜╨║╤Ж╨╕╤П).
             -> [Name]        -- ^ ╨б╨┐╨╕╤Б╨╛╨║ ╨╕╨╝╤С╨╜ ╤Д╤Г╨╜╨║╤Ж╨╕╨╣, ╨║╨╛╤В╨╛╤А╤Л╨╡ ╨▒╤Л╨┤╤Г╤В ╨▓╤Л╤Б╤В╤Г╨┐╨░╤В╤М ╨▓ ╨║╨░╤З╨╡╤Б╤В╨▓╨╡ ╨╛╨▒╤А╨░╨▒╨╛╤В╤З╨╕╨║╨╛╨▓ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓.
                              -- ╨Ъ╨░╨╢╨┤╨░╤П ╨╕╨╖ ╤Д╤Г╨╜╨║╤Ж╨╕╨╣ ╨┤╨╛╨╗╨╢╨╜╨░ ╨╕╨╝╨╡╤В╤М ╨╕╨┤╨╡╨╜╤В╨╕╤З╨╜╤Л╨╣ ╨╜╨░╨▒╨╛╤А ╤В╨╕╨┐╨╛╨▓ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓ ╨╕ ╤Г╨╜╨╕╨║╨░╨╗╤М╨╜╤Л╨╣ ╤В╨╕╨┐ ╨▓╨╛╨╖╨▓╤А╨░╤В╨░.
             -> [InstMakerQ]  -- ^ ╨б╨┐╨╕╤Б╨╛╨║ ╨│╨╡╨╜╨╡╤А╨░╤В╨╛╤А╨╛╨▓ ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╨╛╨▓ ╨┤╨╗╤П "╨╜╨╡╤П╨▓╨╜╨╛╨│╨╛ ╨┐╤А╨╕╨▓╨╡╨┤╨╡╨╜╨╕╤П" ╨╖╨░╨┤╨░╨╜╨╜╤Л╤Е ╤В╨╕╨┐╨╛╨▓ ╨▓ ╨║╨╗╨░╤Б╤Б╤Г ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓ 
                              -- ╤Б╨╛╨╖╨┤╨░╨▓╨░╨╡╨╝╨╛╨╣ ╤Д╤Г╨╜╨║╤Ж╨╕╨╕
             -> DecsQ         -- ^ ╨б╨┐╨╕╤Б╨╛╨║ ╨╜╨╡╨╛╨▒╤Е╨╛╨┤╨╕╨╝╤Л╤Е ╨┤╨╡╨║╨╗╨░╤А╨░╤Ж╨╕╨╣ ╨╕ ╨╛╨┐╤А╨╡╨┤╨╡╨╗╨╡╨╜╨╕╨╣ ╨║╨╗╨░╤Б╤Б╨╛╨▓, ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╨╛╨▓ ╨╕ ╤Д╤Г╨╜╨║╤Ж╨╕╨╣
defVargsFun' fn srcFns sts = do
    -- ╨У╨╡╨╜╨╡╤А╨░╤Ж╨╕╤П ╨╕╨╝╤С╨╜ ╨╜╨╡╨╛╨▒╤Е╨╛╨┤╨╕╨╝╤Л╤Е ╨║╨╗╨░╤Б╤Б╨╛╨▓ ╨╕ ╨╝╨╡╤В╨╛╨┤╨╛╨▓
    argPrc <- nn "ArgPrc" -- ╨Ю╤Б╨╜╨╛╨▓╨╜╨╛╨╣ ╨║╨╗╨░╤Б╤Б ╨┤╨╗╤П ╨╛╨▒╤А╨░╨╛╨▒╤В╨║╨╕ ╨┐╨╡╤А╨╡╨╝╨╡╨╜╨╜╨╛╨│╨╛ ╤З╨╕╤Б╨╗╨░ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓ (╨┤╨░╨╗╨╡╨╡, ╨Ъ╨╗╨░╤Б╤Б).
    argSrc <- nn "ArgSrc" -- ╨Ъ╨╗╨░╤Б╤Б ╨┤╨╗╤П ╤В╨╕╨┐╨╛╨▓, ╨┤╨╛╨┐╤Г╤Б╨║╨░╤О╤Й╨╕╤Е╤Б╤П ╨▓ ╨║╨░╤З╨╡╤Б╤В╨▓╨╡ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓ ╨д╤Г╨╜╨║╤Ж╨╕╨╕ (╨┤╨░╨╗╨╡╨╡, ╨Ш╤Б╤В╨╛╤З╨╜╨╕╨║).
    prc    <- nn "prc"    -- ╨Ъ╨╗╤О╤З╨╡╨▓╨╛╨╣ (╨╕ ╨╡╨┤╨╕╨╜╤Б╤В╨▓╨╡╨╜╨╜╤Л╨╣) ╨╝╨╡╤В╨╛╨┤ ╨Ъ╨╗╨░╤Б╤Б╨░.
    acc    <- nn "acc"    -- ╨Я╨░╤А╨░╨╝╨╡╤В╤А-╨╜╨░╨║╨╛╨┐╨╕╤В╨╡╨╗╤М (╤Б╨┐╨╕╤Б╨╛╨║) ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓.
    toArg  <- nn "toArg"  -- ╨Ъ╨╗╤О╤З╨╡╨▓╨╛╨╣ (╨╕ ╨╡╨┤╨╕╨╜╤Б╤В╨▓╨╡╨╜╨╜╤Л╨╣) ╨╝╨╡╤В╨╛╨┤ ╨Ш╤Б╤В╨╛╤З╨╜╨╕╨║╨░: ╤Б╨╛╨╖╨┤╨░╨╜╨╕╨╡ ╨╖╨╜╨░╤З╨╡╨╜╨╕╤П 
                          -- ╤В╨╕╨┐╨░ ╨н╨╗╨╡╨╝╨╡╨╜╤В (╤Б╨╝. ╨╜╨╕╨╢╨╡) ╨╕╨╖ ╨╖╨╜╨░╤З╨╡╨╜╨╕╤П ╤В╨╕╨┐╨░ ╨║╨╗╨░╤Б╤Б╨░ ╨Ш╤Б╤В╨╛╤З╨╜╨╕╨║
        -- ╨Ф╨╗╤П ╨┐╨╡╤А╨▓╨╛╨╣ ╤Д╤Г╨╜╨║╤Ж╨╕╨╕ - ╨╛╤Б╤В╨░╨╗╤М╨╜╤Л╨╡ ╨┤╨╛╨╗╨╢╨╜╤Л ╨╛╤В╨╗╨╕╤З╨░╤В╤М╤Б╤П ╤В╨╛╨╗╤М╨║╨╛ ╤В╨╕╨┐╨╛╨╝ ╨▓╨╛╨╖╨▓╤А╨░╤В╨░
    funs@((t,      -- ╨╕╤Б╤Е╨╛╨┤╨╜╤Л╨╣ ╤В╨╕╨┐ ╤Д╤Г╨╜╨║╤Ж╨╕╨╕ ╤Б ╤В╨╕╨┐╨╛╨╝ ╨▓╨╛╨╖╨▓╤А╨░╤В╨░ ╨╖╨░╨╝╨╡╨╜╤С╨╜╨╜╤Л╨╝ ╨╜╨░ @a@                                              
           t',     -- ╨╕╤Б╤Е╨╛╨┤╨╜╤Л╨╣ ╤В╨╕╨┐ ╤Д╤Г╨╜╨║╤Ж╨╕╨╕ ╨▒╨╡╨╖ ╨┐╨╛╤Б╨╗╨╡╨┤╨╜╨╡╨│╨╛ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨░ (╤Б╨┐╨╕╤Б╨║╨░) ╨╕ ╤Б ╤В╨╕╨┐╨╛╨╝ ╨▓╨╛╨╖╨▓╤А╨░╤В╨░ ╨╖╨░╨╝╨╡╨╜╤С╨╜╨╜╤Л╨╝ ╨╜╨░ @a@
           rt,     -- ╤А╨╡╨╖╤Г╨╗╤М╤В╨╕╤А╤Г╤О╤Й╨╕╨╣ ╤В╨╕╨┐ ╤Д╤Г╨╜╨║╤Ж╨╕╨╕                                                                             
           cnt_at, -- ╤В╨╕╨┐ ╤Н╨╗╨╡╨╝╨╡╨╜╤В╨╛╨▓ ╤Б╨┐╨╕╤Б╨║╨░ ╨┤╨╗╤П ╤В╨╕╨┐╨░ ╨┐╨╛╤Б╨╗╨╡╨┤╨╜╨╡╨│╨╛ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨░ (╨┤╨░╨╗╨╡╨╡, ╨н╨╗╨╡╨╝╨╡╨╜╤В)
           nms     -- ╤Б╨┐╨╕╤Б╨╛╨║ ╨╕╨╝╤С╨╜ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓ ╨▓╨╕╨┤╨░ a1, a2, ... 
           ) : _) <- sequenceQ $ map splitType srcFns -- ╨а╨░╨╖╨▒╨╕╤А╨░╨╡╨╝ ╤В╨╕╨┐╤Л ╤Д╤Г╨╜╨║╤Ж╨╕╨╣ - ╤Д╨╕╨╜╨░╨╗╤М╨╜╤Л╤Е ╨╛╨▒╤А╨░╨▒╨╛╤В╤З╨╕╨║╨╛╨▓ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓
    -- ╨Ш╤Б╨┐╨╛╨╗╤М╨╖╤Г╨╡╨╝╤Л╨╡ ╤В╨╕╨┐╤Л
    let cnt_a  = ConT nm_a
        vrt_a  = VarT nm_a
        vrt_r  = VarT nm_r
        cnt_ap = ConT argPrc
        cnt_as = ConT argSrc
    -- ╨Ш╨╖╨▓╨╗╨╡╨║╨░╨╡╨╝ ╨│╨╡╨╜╨╡╤А╨░╤В╨╛╤А╤Л ╨╕╨╖ ╨╝╨╛╨╜╨░╨┤╤Л Q
    sts'  <- sequenceQ sts 
    -- ╨Ч╨░╨┐╤Г╤Б╨║╨░╨╡╨╝ ╨│╨╡╨╜╨╡╤А╨░╤В╨╛╤А╤Л ╨┤╨╗╤П ╨┐╨╛╤Б╤В╤А╨╛╨╡╨╜╨╕╤П ╨┤╨╡╨║╨╗╨░╤А╨░╤Ж╨╕╨╣ ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╨╛╨▓ ╨║╨╗╨░╤Б╤Б╨░ ╨Ш╤Б╤В╨╛╤З╨╜╨╕╨║
    insts <- sequenceQ $ map (\i -> i cnt_as cnt_at toArg) sts' 
    -- ╨Т╨╛╨╖╨▓╤А╨░╤Й╨░╨╡╨╝ ╤Б╨┐╨╕╤Б╨╛╨║ ╨▓╤Б╨╡╤Е ╨╜╨╡╨╛╨▒╤Е╨╛╨┤╨╕╨╝╤Л╤Е ╨┤╨╡╨║╨╗╨░╤А╨░╤Ж╨╕╨╣
    return $ -- ╨Ъ╨╗╨░╤Б╤Б
             [cls argPrc [SigD prc t]] ++
             -- ╨Ъ╨╗╨░╤Б╤Б ╨┤╨╗╤П ╤В╨╕╨┐╨╛╨▓, ╨┤╨╛╨┐╤Г╤Б╨║╨░╤О╤Й╨╕╤Е╤Б╤П ╨▓ ╨║╨░╤З╨╡╤Б╤В╨▓╨╡ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓ ╨д╤Г╨╜╨║╤Ж╨╕╨╕ (╨┤╨░╨╗╨╡╨╡, ╨Ш╤Б╤В╨╛╤З╨╜╨╕╨║)
             [cls argSrc [SigD toArg (AppT (AppT ArrowT vrt_a) cnt_at)]] ++
             -- ╨н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╤Л ╨Ъ╨╗╨░╤Б╤Б╨░ ╨┤╨╗╤П ╤В╨╡╤А╨╝╨╕╨╜╨░╨╗╤М╨╜╤Л╤Е ╤В╨╕╨┐╨╛╨▓: ╨║╨╗╤О╤З╨╡╨▓╨╛╨╣ ╨╝╨╡╤В╨╛╨┤ (prc_xxx) ╨▓╤Л╨╖╤Л╨▓╨░╨╡╤В ╨╛╨┤╨╜╤Г ╨╕╨╖ 
             -- ╤Д╤Г╨╜╨║╤Ж╨╕╨╣ srcFns (╤В╤Г, ╤В╨╕╨┐ ╨▓╨╛╨╖╨▓╤А╨░╤В╨░ ╨║╨╛╤В╨╛╤А╨╛╨╣ ╤Д╨╕╨│╤Г╤А╨╕╤А╤Г╨╡╤В ╨▓ ╤Н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╨╡)
             [inst [] (AppT cnt_ap rt') 
                    [FunD prc [Clause (map VarP nms')
                        (NormalB $ conv [foldl AppE (VarE srcFn) (map VarE nms'), VarE 'reverse]) []]] 
                | (srcFn, (_, _, rt', _, nms')) <- zip srcFns funs] ++
             -- ╨н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А ╨Ъ╨╗╨░╤Б╤Б╨░ ╨┤╨╗╤П "╨╝╨░╨│╨╕╨╕" - ╨╜╨░╨║╨╛╨┐╨╗╨╡╨╜╨╕╤П ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓ ╤В╨╕╨┐╨╛╨▓ ╨║╨╗╨░╤Б╤Б╨░ ╨Ш╤Б╤В╨╛╤З╨╜╨╕╨║
             [inst [AppT cnt_as vrt_a, AppT cnt_ap vrt_r]  (AppT cnt_ap (AppT (AppT ArrowT vrt_a) vrt_r))  
                    [FunD prc [Clause (map VarP nms ++ [VarP acc])
                        (NormalB $ conv [foldl AppE (VarE prc) (map VarE nms), conv [consTo acc, VarE toArg]]) []]]] ++
             -- ╨н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А ╨║╨╗╨░╤Б╤Б╨░ ╨Ш╤Б╤В╨╛╤З╨╜╨╕╨║ ╨┤╨╗╤П ╤Б╨░╨╝╨╛╨│╨╛ ╤В╨╕╨┐╨░ ╨н╨╗╨╡╨╝╨╡╨╜╤В
             [inst [] (AppT cnt_as cnt_at) [ValD (VarP toArg) (NormalB (VarE 'id)) []]] ++
             -- ╨н╨║╨╖╨╡╨╝╨┐╨╗╤П╤А╤Л ╨║╨╗╨░╤Б╤Б╨░ ╨Ш╤Б╤В╨╛╤З╨╜╨╕╨║ ╨┤╨╗╤П ╨▓╤Б╨╡╤Е ╨╖╨░╨┐╤А╨╛╤И╨╡╨╜╨╜╤Л╤Е ╤В╨╕╨┐╨╛╨▓
             concat insts ++
             -- ╨Ф╨╡╨║╨╗╨░╤А╨░╤Ж╨╕╤П ╨╕ ╨╛╨┐╤А╨╡╨┤╨╡╨╗╨╡╨╜╨╕╨╡ ╨д╤Г╨╜╨║╤Ж╨╕╨╕
             [SigD nm $ ForallT [ptv_a] [AppT cnt_ap vrt_a] t',
              FunD nm [Clause (map VarP nms) (NormalB $ foldl1 AppE ([VarE prc] ++ map VarE nms ++ [ListE []])) []]] 

        where nn pref = newName $ pref ++ "_"  ++ fn
              -- C╤В╤А╨╛╨╕╤В ╨╕╨╖ ╤Б╨┐╨╕╤Б╨║╨░ [Exp] ╨║╨╛╨╜╨▓╨╡╨╡╤А: exp1 . exp2 . ...
              conv    = foldr1 $ \x r -> InfixE (Just x) (VarE '(.)) (Just r) 
              cls c   = ClassD [] c [ptv_a] []
              inst    = InstanceD Nothing 
              consTo  = InfixE Nothing (ConE '(:)) . Just . VarE 
              nm      = mkName fn
              nm_a    = mkName "a"
              ptv_a   = PlainTV nm_a
              nm_r    = mkName "r"
              -- ╨а╨░╨╖╨▒╨╕╨╡╨╜╨╕╨╡ ╤В╨╕╨┐╨░ ╤Д╤Г╨╜╨║╤Ж╨╕╨╕ ╨▓╨╕╨┤╨░ T1 -> T2 -> ... -> [A] -> R ╨╜╨░ ╤Б╨╛╤Б╤В╨░╨▓╨╜╤Л╨╡ ╤Н╨╗╨╡╨╝╨╡╨╜╤В╤Л, ╨│╨╡╨╜╨╡╤А╨░╤Ж╨╕╤П ╨▓╤Б╨┐╨╛╨╝╨╛╨│╨░╤В╨╡╨╗╤М╨╜╤Л╤Е ╨╕╨╝╤С╨╜
              splitType 
                  :: Name -- ^ ╨Ш╨╝╤П ╤Д╤Г╨╜╨║╤Ж╨╕╨╕
                  -> Q (Type, Type, Type, Type, [Name]) 
                  -- ^ * ╨╕╤Б╤Е╨╛╨┤╨╜╤Л╨╣ ╤В╨╕╨┐ ╤Д╤Г╨╜╨║╤Ж╨╕╨╕ ╤Б ╤В╨╕╨┐╨╛╨╝ ╨▓╨╛╨╖╨▓╤А╨░╤В╨░ ╨╖╨░╨╝╨╡╨╜╤С╨╜╨╜╤Л╨╝ ╨╜╨░ @a@
                  -- * ╨╕╤Б╤Е╨╛╨┤╨╜╤Л╨╣ ╤В╨╕╨┐ ╤Д╤Г╨╜╨║╤Ж╨╕╨╕ ╨▒╨╡╨╖ ╨┐╨╛╤Б╨╗╨╡╨┤╨╜╨╡╨│╨╛ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨░ (╤Б╨┐╨╕╤Б╨║╨░) ╨╕ ╤Б ╤В╨╕╨┐╨╛╨╝ ╨▓╨╛╨╖╨▓╤А╨░╤В╨░ ╨╖╨░╨╝╨╡╨╜╤С╨╜╨╜╤Л╨╝ ╨╜╨░ @a@
                  -- * ╤В╨╕╨┐ ╨▓╨╛╨╖╨▓╤А╨░╤В╨░ ╤Д╤Г╨╜╨║╤Ж╨╕╨╕
                  -- * ╤В╨╕╨┐ ╤Н╨╗╨╡╨╝╨╡╨╜╤В╨╛╨▓ ╤Б╨┐╨╕╤Б╨║╨░ ╨┤╨╗╤П ╤В╨╕╨┐╨░ ╨┐╨╛╤Б╨╗╨╡╨┤╨╜╨╡╨│╨╛ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨░: ╤В╨╕╨┐ A ╨┤╨╗╤П ╤Д╤Г╨╜╨║╤Ж╨╕ ╨▓╨╕╨┤╨░ T1 -> T2 -> ... -> [A] -> R
                  -- * ╤Б╨┐╨╕╤Б╨╛╨║ ╨╕╨╝╤С╨╜ ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓ ╨▓╨╕╨┤╨░ a1, a2, ... ╨║╨╛╨╗╨╕╤З╨╡╤Б╤В╨▓╨╛╨╝ ╤А╨░╨▓╨╜╤Л╨╝ ╤З╨╕╤Б╨╗╤Г ╨┐╨░╤А╨░╨╝╨╡╤В╨╛╨▓ ╨▓ ╨╕╤Б╤Е╨╛╨┤╨╜╨╛╨╝ ╤В╨╕╨┐╨╡-╤Д╤Г╨╜╨║╤Ж╨╕╨╕ - 1
                  --
                  -- ╨Я╤А╨╕╨╝╨╡╤А:
                  --
                  --  ╨┤╨╗╤П ╤Д╤Г╨╜╨║╤Ж╨╕╨╕ ╤В╨╕╨┐╨░: Int -> String -> [Double] -> IO () 
                  --  ╨┐╨╛╨╗╤Г╤З╨╕╨╝: ( Int -> String -> [Double] -> a, Int -> String -> a, IO (), Double. ['a1, 'a2] )
              splitType nm = do
                  VarI _ rt _ <- reify nm
                  let tps = funTypes rt
                      t   = mkType . init $ tps
                      t'  = mkType . init . init $ tps
                  return (t, t', last tps, 
                          let AppT ArrowT (AppT ListT at) = last . init $ tps in at, -- ╨▓╤Л╨┤╨╡╨╗╤П╨╡╨╝ ╤В╨╕╨┐ ╤Н╨╗╨╡╨╝╨╡╨╜╤В╨░ ╤Б╨┐╨╕╤Б╨║╨░
                          map mkName ["a" ++ show i | i <- [1 .. length tps - 2]]) 
                      where -- | ╨б╨╛╨▒╨╕╤А╨░╨╡╨╝ ╨╛╨▒╤А╨░╤В╨╜╨╛ ╤Д╤Г╨╜╨║╤Ж╨╕╨╛╨╜╨░╨╗╤М╨╜╤Л╨╣ ╤В╨╕╨┐ ╨╕╨╖ ╤Б╨┐╨╕╤Б╨║╨░ ╤В╨╕╨┐╨╛╨▓ ╤Н╨╗╨╡╨╝╨╡╨╜╤В╨╛╨▓ 
                            -- ╨╕ ╤А╨╡╨╖╤Г╨╗╤М╤В╨░╤В╨░ (╨╛╨▒╤А╨░╤В╨╜╨░╤П ╤Д╤Г╨╜╨║╤Ж╨╕╤П ╨┤╨╗╤П funTypes).
                            --
                            -- ╨Я╤А╨╕╨╝╨╡╤А (╨┐╤Б╨╡╨▓╨┤╨╛╨║╨╛╨┤): mkType [a, b-> c, [d], r] ==> a -> (b -> c) -> [d] -> r
                            mkType = foldr1 AppT . (++ [VarT $ mkName "a"])
                            -- | ╨а╨░╨╖╨▒╨╕╨▓╨░╨╡╨╝ ╤В╨╕╨┐ ╤Д╤Г╨╜╨║╤Ж╨╕╨╕ ╨╜╨░ ╤В╨╕╨┐╤Л ╨┐╨░╤А╨░╨╝╨╡╤В╤А╨╛╨▓ ╨╕ ╤А╨╡╨╖╤Г╨╗╤М╤В╨░╤В╨░.
                            --
                            -- ╨Я╤А╨╕╨╝╨╡╤А (╨┐╤Б╨╡╨▓╨┤╨╛╨║╨╛╨┤): funTypes a -> (b -> c) -> [d] -> r ==> [a, b-> c, [d], r]
                            funTypes (AppT a@(AppT ArrowT _) x) = a : funTypes x
                            funTypes x = [x]
